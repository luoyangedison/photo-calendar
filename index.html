-- 用户表 CREATE TABLE users ( id SERIAL PRIMARY KEY, username TEXT UNIQUE NOT NULL, password_hash TEXT NOT NULL, created_at TIMESTAMPTZ DEFAULT NOW() ); ALTER TABLE users ENABLE ROW LEVEL SECURITY; CREATE POLICY "users_public" ON users FOR SELECT USING (true); CREATE POLICY "users_insert" ON users FOR INSERT WITH CHECK (true); -- 事件表 CREATE TABLE events ( id TEXT PRIMARY KEY, date TEXT NOT NULL, period TEXT DEFAULT 'morning', type TEXT DEFAULT 'shooting', title TEXT NOT NULL, customer_name TEXT DEFAULT '', customer_phone TEXT DEFAULT '', assignee TEXT DEFAULT '', description TEXT DEFAULT '', reminder_enabled BOOLEAN DEFAULT false, reminder_minutes INTEGER DEFAULT 0, reminder_notified BOOLEAN DEFAULT false, created_by TEXT DEFAULT '', created_at TEXT DEFAULT '', time TEXT DEFAULT '' ); ALTER TABLE events ENABLE ROW LEVEL SECURITY; CREATE POLICY "events_public" ON events FOR ALL USING (true); -- 登录验证函数 CREATE OR REPLACE FUNCTION verify_user(uname TEXT, phash TEXT) RETURNS BOOLEAN AS $$ SELECT EXISTS(SELECT 1 FROM users WHERE username=uname AND password_hash=phash); $$ LANGUAGE sql SECURITY DEFINER;
